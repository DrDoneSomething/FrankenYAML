Lists are integrations that only include entities stored in list format, such as automation and switch.

Ex: switch
switch:
  - platform: mqtt
    name: "Casita AC"
    state_topic: "stat/SNF-CasitaAC/POWER"
    command_topic: "cmnd/SNF-CasitaAC/POWER"
    availability_topic: "tele/SNF-CasitaAC/LWT"
    retain: true
  - platform: mqtt
    name: "Carport Light"
    state_topic: "stat/carportnkitchenentryway/POWER1"
    command_topic: "cmnd/carportnkitchenentryway/POWER1"
    availability_topic: "tele/carportnkitchenentryway/LWT"
    qos: 1
    retain: false
